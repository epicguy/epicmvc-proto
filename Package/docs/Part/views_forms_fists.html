Forms and FISTs (Field Lists): Creating and Editing

Forms and Fields are typically created in the application's fist.coffee file. Fields are defined independent of a named Fist, and later, a Fist is made by simply listing the various Fields that make up the Form in the order to display them. For any field, you specify a plethora of information designed to provide all the processing necessary for filtering between what the user sees and what database or REST Api will ultimately store/receive as data for those fields.

example/fist.coffee coffeescript 
E.fist$example=
	FIELDS:
		LoginName: db_nm:'username', type:'text',     req:true, h2h:'trim:lower', label:'Username'
		LoginPass: db_nm:'password', type:'password', req:true, h2h:'trim', label:'Password'

		CreatePass:
			db_nm:'password', type:'password', req:true, h2h:'trim', label:'Password',
			validate: 'pwd_strength', issue_text: 'Password must be at least 6 characters and have 1 or more number(s).
		ConfirmPass:
			db_nm:'confirm_pass', type:'password', h2h:'trim', label:'Confirm Password', req:true
			confirm: 'CreatePass', # The field to match when that field is not empty and not at issue
			issue_text: 'Passwords do not match. Please re-enter.  Passwords are case sensitive.'

		Email:     db_nm:'email',    type: 'text', label: 'Email', validate:'email'
		Gender:
			db_nm:'genderTk',type:'pulldown:array', label:'Gender', choice: 'Select Gender'
			cdata: [  ['M','Male'],  ['F','Female']  ]

		CcCvv:
			db_nm:'cc_cvv', type:'text', req:true, label: 'CVV'
			h2h:'digits', validate: 'test', validate_expr: '[0-9]{3,4}'
			req_text: 'CVV is required'
			issue_text: 'CVV must be 3 or 4 digits (check the back of your card)'
		CcMonth:
			db_nm:'cc_month', type:'pulldown:array', req:true, label: 'Month',
			cdata: ['01', '02', '03', '04', '05', '06', '07', '08', '09', '10', '11', '12' ]
		CcYear:
			db_nm:'cc_year', type:'pulldown:array', req:true, label: 'Year', default: '2014'
			cdata: ['2014', '2015', '2016', '2017', '2018', '2019', '2020', '2021', '2022', '2023', '2024']
		CcZip:
			db_nm:'cc_zip', type:'text', req:true, label: 'Zip Code'
			validate:'test', validate_expr: '^[0-9]{5}(-?[0-9]{4})?$'
			issue_text: 'Zip must be 5 or 9 digits'

		PrivayPolicy:
			db_nm:'privacy_policy', type:'yesno', cdata:['yes','no], label: 'Agree to Privacy Policy?', req: true }
		Terms:
			db_nm:'terms_conditions', type:'yesno', cdata:['yes','no], label: 'Agree to Terms & Conditions?', req: true }

		Timezone:
			db_nm:'timezone', type: 'pulldown:wist', cdata: 'timezone:token:nice', label: 'Timezone'
			choice: 'Select Timezone'
		AskInclude:
			db_nm:'include_children', type:'radio:array', req:true, label: 'Options:'
			cdata: [[0,'Parents Only'], [1,'Include Children']], default: 0
			h2d: 'blank', d2h: 'zero'

	FISTS:

		Login:		FIELDS: [ 'LoginName', 'LoginPass' ], part: 'fist_login', via: 'Field'
		SignUp:		FIELDS: [ 'CreatePass', 'ConfirmPass' ]
		CreditCard:	FIELDS: [ 'CcCvv', 'CcMonth', 'CcYear', 'CcZip']
		Legal		FIELDS: [ 'PrivacyPolicy', 'Terms' ]
		Profile:	FIELDS: [ ..., 'Timezone', ... ]
